```typescript TypeScript
import { Morph } from "@runmorph/cloud";

  userContext ? JSON.stringify(userContext) : "empty";

// Initialize Morph instance
const morph = Morph({
  apiKey: process.env.MORPH_API_KEY,
  apiSecret: process.env.MORPH_API_SECRET,
});

// Create a connection
const { data, error } = await morph
  .connection({
    connectorId: "hubspot",
    ownerId: "my_customer_id",
  })
  .create({ operations: ["genericContact::list", "crmOpportunity::create"] });
```
